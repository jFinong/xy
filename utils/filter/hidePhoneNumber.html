    # 隐藏手机号中间四位

    隐藏手机号中间四位。

    # 示例：

    177****8767

    # 代码实现：

    ...
<div>{{phoneNumber | hidePhoneMiddle}}</div>
...

<script>
import { hidePhoneMiddle } from "@xiyun/utils";

export default {
    filters: { hidePhoneMiddle },
    data() {
        return {
            phoneNumber: '17789098767',
        }
    },
}
</script>
